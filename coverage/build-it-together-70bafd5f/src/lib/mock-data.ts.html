export function getCrowdColor(pct: number): 'green' | 'amber' | 'red' {
  if (pct < 0.5) return 'green';
  if (pct < 0.8) return 'amber';
  return 'red';
}
 
export function getCrowdLabel(pct: number): string {
  if (pct < 0.5) return 'Clear';
  if (pct < 0.8) return 'Moderate';
  return 'Crowded';
}
